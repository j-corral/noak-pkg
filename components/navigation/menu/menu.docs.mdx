### Component usage

```js
import { Menu } from '@hoomies/noak.components.navigation.menu';

const routes = [
  {
    literal: true,
    label: 'Home',
    path: '/',
    icon: '',
  },
  {
    label: 'menu:page.legal',
    path: '/legal',
    lang: { fr: '/mentions-legales', en: '/en/legal-notice' },
  },
];
<Menu routes={routes} />;
```
