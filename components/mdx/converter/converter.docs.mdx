import { MDXConverter, MDXConverterProps } from './converter';

## Description

Convert serialized MDX (with ParseMDX) into a React component (Thanks to MDXRemote)

## Component usage

### Inside the Next Page

```js
import { MDXConverter, MDXConverterProps } from '@hoomies/noak.components.mdx.converter';

const serializedMDX = {...};

<Converter source={serializedMDX} />
```

### Next PAge Example

```js
import useTranslation from 'next-translate/useTranslation';

import type { NextPage } from '@hoomies/noak.types.next';
import { ParseMDX } from '@hoomies/fuel.functions.parse-mdx';
import { MDXConverter, MDXConverterProps } from '@hoomies/noak.components.mdx.converter';

const LegalPage: NextPage<MDXConverterProps> = ({ source, componentNames }) => {
  const { t } = useTranslation();

  return <>{<MDXConverter source={source} componentNames={componentNames} />}</>;
};

export default LegalPage;

export async function getStaticProps({}) {
  const source = `---
title: Front Matter
---

<Unagui.Hero title={title} />

  `;

  const MDXProps = await ParseMDX({ source });

  return { props: { ...MDXProps } };
}
```

### Inside the getStaticProps method

PLease refer to [ParseMDX Function](https://bit.dev/hoomies/fuel/functions/parse-mdx).
