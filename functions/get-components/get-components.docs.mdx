import { GetComponents } from './get-components';

## Description

Return an object containing Selected components (Noak + Unagui + MDX)

### Function sage

```js
import { GetComponents } from '@hoomies/noak.functions.get-components';

const selectedComponents = GetComponents({ names: ['UnaguiHero', 'UnaguiSection', 'NoakLocaleSwitcher'] });
// result = { h1: [Function: h1], p: [Function: p], ...,'Unagui.Hero': [Function], 'Unagui.Section': [Function], 'Noak.LocaleSwitcher': [Function]}

const withoutMDX = GetComponents({ names: ['UnaguiHero', 'UnaguiSection'], withMDX: false });
// result = { 'UnaguiHero': [Function], 'UnaguiSection': [Function]}
```
